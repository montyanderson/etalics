etalics

a generative real-time graphics library for artists

etalics("#one",
	(x, y, i) => [
		Math.sin(x + y + i / 10),
		Math.sin(x + y + i / 10),
		Math.sin(x + y + i / 10)
	]
);
		
etalics("#two",
	(x, y, i) => [
		Math.sin(x - y + i / 10),
		Math.sin(x - y + i / 10),
		Math.sin(x + y + i / 10)
	]
);
		
etalics("#three",
	(x, y, i) => [
		Math.sin(Math.cosh(x + y + (24 - (i % 200)) / 10)),
		Math.sin(Math.cosh(100 + x - y - (24 - (i % 200)) / 10)),
		Math.sin(Math.cosh(x + y + (24 - (i % 200)) / 10))
	]
);
